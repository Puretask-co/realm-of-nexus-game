javascript
// src/systems/DSPManager.js
class DSPManager {
  constructor() {
    this.currentDSP = 100;  // Starts full
    this.maxDSP = 100;
    this.criticalThreshold = 20;  // Warning zone
    this.emergencyThreshold = 5;   // Danger zone
    this.regenRate = 0;  // DSP doesn't naturally regen
    this.listeners = [];  // For UI updates
  }
  
  consume(amount, spellName) {
    if (this.currentDSP < amount) {
      return { success: false, reason: "Insufficient DSP" };
    }
    
    this.currentDSP -= amount;
    this.notifyListeners();
    
    // Log for debugging
    console.log(`${spellName} consumed ${amount} DSP. Remaining: ${this.currentDSP}`);
    
    // Check thresholds
    if (this.currentDSP <= this.emergencyThreshold) {
      this.triggerEmergencyWarning();
    } else if (this.currentDSP <= this.criticalThreshold) {
      this.triggerCriticalWarning();
    }
    
    return { success: true, newDSP: this.currentDSP };
  }
  
  restore(amount, source) {
    this.currentDSP = Math.min(this.currentDSP + amount, this.maxDSP);
    this.notifyListeners();
    console.log(`${source} restored ${amount} DSP. Current: ${this.currentDSP}`);
  }
  
  notifyListeners() {
    this.listeners.forEach(callback => callback(this.currentDSP, this.maxDSP));
  }
  
  subscribe(callback) {
    this.listeners.push(callback);
  }
}

// src/ui/DSPBar.js
class DSPBar {
  constructor(scene, x, y) {
    this.scene = scene;
    
    // Background bar (always visible)
    this.bgBar = scene.add.rectangle(x, y, 300, 30, 0x2d2d2d);
    this.bgBar.setStrokeStyle(2, 0x1a1a1a);
    this.bgBar.setScrollFactor(0);  // Fixed to camera
    
    // Fill bar (shows current DSP)
    this.fillBar = scene.add.rectangle(x - 148, y, 296, 26, 0x3CFFD9);
    this.fillBar.setOrigin(0, 0.5);
    this.fillBar.setScrollFactor(0);
    
    // DSP text
    this.text = scene.add.text(x, y, '100 / 100', {
      fontSize: '14px',
      color: '#FFFFFF',
      fontFamily: 'Arial',
      fontStyle: 'bold'
    });
    this.text.setOrigin(0.5);
    this.text.setScrollFactor(0);
    
    // Label
    this.label = scene.add.text(x - 160, y - 25, 'Deep Sap Pool', {
      fontSize: '12px',
      color: '#AAAAAA',
      fontFamily: 'Arial'
    });
    this.label.setScrollFactor(0);
    
    // Warning indicators (hidden by default)
    this.criticalWarning = scene.add.text(x, y + 25, '⚠ CRITICAL DSP', {
      fontSize: '12px',
      color: '#FF6B35',
      fontFamily: 'Arial',
      fontStyle: 'bold'
    });
    this.criticalWarning.setOrigin(0.5);
    this.criticalWarning.setScrollFactor(0);
    this.criticalWarning.setVisible(false);
    
    this.emergencyWarning = scene.add.text(x, y + 25, '🔴 EMERGENCY - VERDANCE DYING', {
      fontSize: '14px',
      color: '#FF0000',
      fontFamily: 'Arial',
      fontStyle: 'bold'
    });
    this.emergencyWarning.setOrigin(0.5);
    this.emergencyWarning.setScrollFactor(0);
    this.emergencyWarning.setVisible(false);
    
    // Pulsing animation for warnings
    this.warningTween = null;
  }
  
  update(currentDSP, maxDSP) {
    // Update fill bar width
    const percentage = currentDSP / maxDSP;
    this.fillBar.width = 296 * percentage;
    
    // Update color based on percentage
    if (percentage <= 0.05) {
      this.fillBar.setFillStyle(0xFF0000);  // Red (emergency)
      this.showEmergencyWarning();
    } else if (percentage <= 0.20) {
      this.fillBar.setFillStyle(0xFF6B35);  // Orange (critical)
      this.showCriticalWarning();
    } else {
      this.fillBar.setFillStyle(0x3CFFD9);  // Teal (healthy)
      this.hideWarnings();
    }
    
    // Update text
    this.text.setText(`${Math.floor(currentDSP)} / ${maxDSP}`);
  }
  
  showCriticalWarning() {
    this.criticalWarning.setVisible(true);
    this.emergencyWarning.setVisible(false);
    
    // Pulse animation
    if (!this.warningTween || !this.warningTween.isPlaying()) {
      this.warningTween = this.scene.tweens.add({
        targets: this.criticalWarning,
        alpha: 0.3,
        duration: 500,
        yoyo: true,
        repeat: -1
      });
    }
  }
  
  showEmergencyWarning() {
    this.criticalWarning.setVisible(false);
    this.emergencyWarning.setVisible(true);
    
    // Faster pulse for emergency
    if (!this.warningTween || !this.warningTween.isPlaying()) {
      this.warningTween = this.scene.tweens.add({
        targets: this.emergencyWarning,
        alpha: 0,
        duration: 300,
        yoyo: true,
        repeat: -1
      });
    }
  }
  
  hideWarnings() {
    this.criticalWarning.setVisible(false);
    this.emergencyWarning.setVisible(false);
    if (this.warningTween) {
      this.warningTween.stop();
    }
  }
}

      y: 20,
      alpha: 0,
      duration: 1000,
      onComplete: () => text.destroy()
    });
  }

class SpellTargeting {
  constructor(scene) {
    this.scene = scene;
    this.isActive = false;
    this.currentSpell = null;
    this.validTargets = [];
    this.hoveredTarget = null;
    
    // Targeting UI elements
    this.rangeIndicator = null;
    this.aoeIndicator = null;
    this.targetHighlight = null;
    this.cancelText = null;
    
    this.setupTargetingUI();
  }
  
  setupTargetingUI() {
    // Range indicator (circle showing max range)
    this.rangeIndicator = this.scene.add.circle(0, 0, 0, 0x3CFFD9, 0.1);
    this.rangeIndicator.setStrokeStyle(2, 0x3CFFD9, 0.5);
    this.rangeIndicator.setVisible(false);
    this.rangeIndicator.setDepth(100);
    
    // AOE indicator (circle showing spell area of effect)
    this.aoeIndicator = this.scene.add.circle(0, 0, 0, 0xFF6B35, 0.2);
    this.aoeIndicator.setStrokeStyle(2, 0xFF6B35, 0.8);
    this.aoeIndicator.setVisible(false);
    this.aoeIndicator.setDepth(101);
    
    // Target highlight
    this.targetHighlight = this.scene.add.rectangle(0, 0, 48, 48, 0xFFFFFF, 0);
    this.targetHighlight.setStrokeStyle(3, 0xFFFFFF, 1);
    this.targetHighlight.setVisible(false);
    this.targetHighlight.setDepth(102);
    
    // Pulsing animation for highlight
    this.scene.tweens.add({
      targets: this.targetHighlight,
      alpha: 0.3,
      duration: 500,
      yoyo: true,
      repeat: -1
    });
    
    // Cancel instruction
    this.cancelText = this.scene.add.text(400, 550, 'Right-click or press ESC to cancel', {
      fontSize: '14px',
      color: '#FFFFFF',
      backgroundColor: '#000000',
      padding: { x: 10, y: 5 }
    });
    this.cancelText.setOrigin(0.5);
    this.cancelText.setScrollFactor(0);
    this.cancelText.setDepth(1000);
    this.cancelText.setVisible(false);
    
    // Mouse input
    this.scene.input.on('pointermove', (pointer) => {
      if (this.isActive) {
        this.updateTargeting(pointer);
      }
    });
    
    this.scene.input.on('pointerdown', (pointer) => {
      if (this.isActive) {
        if (pointer.leftButtonDown()) {
          this.confirmTarget(pointer);
        } else if (pointer.rightButtonDown()) {
          this.cancel();
        }
      }
    });
    
    // Keyboard cancel
    this.scene.input.keyboard.on('keydown-ESC', () => {
      if (this.isActive) this.cancel();
    });
  }
  
  activate(spell) {
    this.isActive = true;
    this.currentSpell = spell;
    
    console.log(`Targeting mode activated for: ${spell.name}`);
    
    // Show range indicator
    if (spell.range > 0) {
      const player = this.scene.player;
      this.rangeIndicator.setPosition(player.x, player.y);
      this.rangeIndicator.setRadius(spell.range * 50);  // 50px per tile
      this.rangeIndicator.setVisible(true);
    }
    
    // Calculate valid targets based on spell type
    this.calculateValidTargets();
    
    // Show cancel text
    this.cancelText.setVisible(true);
    
    // Change cursor
    this.scene.input.setDefaultCursor('crosshair');
  }
  
  deactivate() {
    this.isActive = false;
    this.currentSpell = null;
    this.validTargets = [];
    this.hoveredTarget = null;
    
    // Hide UI
    this.rangeIndicator.setVisible(false);
    this.aoeIndicator.setVisible(false);
    this.targetHighlight.setVisible(false);
    this.cancelText.setVisible(false);
    
    // Reset cursor
    this.scene.input.setDefaultCursor('default');
  }
  
  calculateValidTargets() {
    this.validTargets = [];
    const spell = this.currentSpell;
    const player = this.scene.player;
    
    switch (spell.targetType) {
      case 'self':
        this.validTargets.push({ type: 'self', target: player });
        break;
        
      case 'ally_single':
        // Find all allies in range
        this.scene.allies.forEach(ally => {
          const distance = Phaser.Math.Distance.Between(
            player.x, player.y, ally.x, ally.y
          ) / 50;  // Convert pixels to tiles
          
          if (distance <= spell.range) {
            this.validTargets.push({ type: 'ally', target: ally });
          }
        });
        break;
        
      case 'enemy_single':
        // Find all enemies in range
        this.scene.enemies.forEach(enemy => {
          const distance = Phaser.Math.Distance.Between(
            player.x, player.y, enemy.x, enemy.y
          ) / 50;
          
          if (distance <= spell.range) {
            this.validTargets.push({ type: 'enemy', target: enemy });
          }
        });
        break;
        
      case 'ground_aoe':
      case 'tile':
        // Any tile in range is valid
        const maxRange = spell.range * 50;
        for (let dx = -maxRange; dx <= maxRange; dx += 50) {
          for (let dy = -maxRange; dy <= maxRange; dy += 50) {
            const distance = Math.sqrt(dx * dx + dy * dy);
            if (distance <= maxRange) {
              this.validTargets.push({
                type: 'ground',
                x: player.x + dx,
                y: player.y + dy
              });
            }
          }
        }
        break;
    }
    
    console.log(`Found ${this.validTargets.length} valid targets`);
  }
  
  updateTargeting(pointer) {
    const worldPoint = this.scene.cameras.main.getWorldPoint(pointer.x, pointer.y);
    
    // Snap to grid
    const gridX = Math.round(worldPoint.x / 50) * 50;
    const gridY = Math.round(worldPoint.y / 50) * 50;
    
    // Check if mouse is over a valid target
    let foundTarget = null;
    
    if (this.currentSpell.targetType === 'enemy_single' || 
        this.currentSpell.targetType === 'ally_single') {
      // Check entities
      for (let targetData of this.validTargets) {
        if (targetData.target) {
          const distance = Phaser.Math.Distance.Between(
            targetData.target.x, targetData.target.y, gridX, gridY
          );
          
          if (distance < 25) {  // Within half a tile
            foundTarget = targetData;
            break;
          }
        }
      }
    } else if (this.currentSpell.targetType === 'ground_aoe' || 
               this.currentSpell.targetType === 'tile') {
      // Check ground tiles
      const player = this.scene.player;
      const distance = Phaser.Math.Distance.Between(
        player.x, player.y, gridX, gridY
      ) / 50;
      
      if (distance <= this.currentSpell.range) {
        foundTarget = { type: 'ground', x: gridX, y: gridY };
      }
    }
    
    // Update highlight
    if (foundTarget) {
      this.hoveredTarget = foundTarget;
      
      const targetX = foundTarget.target ? foundTarget.target.x : foundTarget.x;
      const targetY = foundTarget.target ? foundTarget.target.y : foundTarget.y;
      
      this.targetHighlight.setPosition(targetX, targetY);
      this.targetHighlight.setVisible(true);
      
      // Show AOE if applicable
      if (this.currentSpell.aoeRadius > 0) {
        this.aoeIndicator.setPosition(targetX, targetY);
        this.aoeIndicator.setRadius(this.currentSpell.aoeRadius * 50);
        this.aoeIndicator.setVisible(true);
      }
    } else {
      this.hoveredTarget = null;
      this.targetHighlight.setVisible(false);
      this.aoeIndicator.setVisible(false);
    }
  }
  
  confirmTarget(pointer) {
    if (!this.hoveredTarget) {
      console.log('No valid target selected');
      return;
    }
    
    // Execute spell
    this.scene.executeSpell(this.currentSpell, this.hoveredTarget);
    
    // Deactivate targeting
    this.deactivate();
  }
  
  cancel() {
    console.log('Spell targeting cancelled');
    this.deactivate();
  }
}

      update: (particle) => {
        const age = 1 - (particle.life / particle.lifespan);

        particle.y = y + Math.sin(angle) * radius;

// src/systems/AudioManager.js
class AudioManager {
  constructor(scene) {
    this.scene = scene;
    this.sounds = {};
    this.musicVolume = 0.7;
    this.sfxVolume = 0.8;
    this.isMuted = false;
    
    this.loadAllAudio();
  }
  
  loadAllAudio() {
    // Load all sounds from manifest
    Object.keys(AUDIO_MANIFEST).forEach(key => {
      try {
        this.sounds[key] = this.scene.sound.add(key, {
          volume: this.sfxVolume
        });
      } catch (e) {
        console.warn(`Failed to load audio: ${key}`, e);
        // Create silent placeholder
        this.sounds[key] = { play: () => {}, stop: () => {} };
      }
    });
    
    console.log(`Loaded ${Object.keys(this.sounds).length} sounds`);
  }
  
  play(soundKey, config = {}) {
    if (this.isMuted) return;
    
    const sound = this.sounds[soundKey];
    if (!sound) {
      console.warn(`Sound not found: ${soundKey}`);
      return;
    }
    
    const volume = config.volume !== undefined ? config.volume : this.sfxVolume;
    const loop = config.loop || false;
    const rate = config.rate || 1;  // Playback speed
    
    try {
      if (sound.isPlaying && !config.allowOverlap) {
        sound.stop();
      }
      
      sound.play({
        volume: volume,
        loop: loop,
        rate: rate
      });
    } catch (e) {
      console.warn(`Failed to play sound: ${soundKey}`, e);
    }
  }
  
  stop(soundKey) {
    const sound = this.sounds[soundKey];
    if (sound && sound.isPlaying) {
      sound.stop();
    }
  }
  
  stopAll() {
    Object.values(this.sounds).forEach(sound => {
      if (sound.isPlaying) {
        sound.stop();
      }
    });
  }
  
  setMusicVolume(volume) {
    this.musicVolume = Phaser.Math.Clamp(volume, 0, 1);
    // Apply to music tracks
    // (Will implement when we add music)
  }
  
  setSFXVolume(volume) {
    this.sfxVolume = Phaser.Math.Clamp(volume, 0, 1);
    Object.values(this.sounds).forEach(sound => {
      if (sound.setVolume) {
        sound.setVolume(this.sfxVolume);
      }
    });
  }
  
  toggleMute() {
    this.isMuted = !this.isMuted;
    if (this.isMuted) {
      this.stopAll();
    }
  }
  
  // Helper: play spell cast sound
  playSpellCast(spellId) {
    // Different spells can have different cast sounds
    const spell = SPELL_DATABASE[spellId];
    if (!spell) return;
    
    const soundKey = spell.animation?.soundEffect || 'magic_cast_generic';
    this.play(soundKey);
  }
  
  // Helper: play impact sound
  playSpellImpact(spellId) {
    this.play('impact_magic', { volume: 0.6 });
  }
}

export default AudioManager;

// Add to DSPBar.js

update(currentDSP, maxDSP) {
  const oldPercentage = this.fillBar.width / 296;
  const newPercentage = currentDSP / maxDSP;
  
  // Smooth transition instead of instant change
  this.scene.tweens.add({
    targets: this.fillBar,
    width: 296 * newPercentage,
    duration: 300,
    ease: 'Cubic.easeOut'
  });
  
  // Pulse on significant drain (>10 DSP)
  const dspLost = (oldPercentage - newPercentage) * 100;
  if (dspLost > 10) {
    this.scene.tweens.add({
      targets: this.fillBar,
      scaleY: 1.2,
      duration: 100,
      yoyo: true,
      ease: 'Sine.easeInOut'
    });
  }
  
  // Update color with smooth transition
  const targetColor = this.getColorForPercentage(newPercentage);
  this.fillBar.setFillStyle(targetColor);
  
  // Update text
  this.text.setText(`${Math.floor(currentDSP)} / ${maxDSP}`);
}

getColorForPercentage(percentage) {
  if (percentage <= 0.05) return 0xFF0000;  // Red
  if (percentage <= 0.20) return 0xFF6B35;  // Orange
  if (percentage <= 0.50) return 0xFFBB33;  // Yellow
  return 0x3CFFD9;  // Teal (healthy)
}

// src/systems/CooldownManager.js
class CooldownManager {
  constructor(scene) {
    this.scene = scene;
    this.cooldowns = new Map();  // spellId -> turnsRemaining
    this.listeners = [];
  }
  
  /**
   * Start cooldown for a spell
   * @param {string} spellId - Unique spell identifier
   * @param {number} turns - Number of turns for cooldown
   */
  startCooldown(spellId, turns) {
    if (turns <= 0) return;  // No cooldown needed
    
    this.cooldowns.set(spellId, turns);
    this.notifyListeners(spellId, turns);
    
    console.log(`${spellId} on cooldown for ${turns} turns`);
  }
  
  /**
   * Check if spell is on cooldown
   * @param {string} spellId
   * @returns {boolean}
   */
  isOnCooldown(spellId) {
    return this.cooldowns.has(spellId) && this.cooldowns.get(spellId) > 0;
  }
  
  /**
   * Get remaining cooldown turns
   * @param {string} spellId
   * @returns {number} - 0 if not on cooldown
   */
  getRemainingTurns(spellId) {
    return this.cooldowns.get(spellId) || 0;
  }
  
  /**
   * Reduce all cooldowns by 1 (call at end of turn)
   */
  tick() {
    const expiredSpells = [];
    
    this.cooldowns.forEach((turns, spellId) => {
      const newTurns = turns - 1;
      
      if (newTurns <= 0) {
        this.cooldowns.delete(spellId);
        expiredSpells.push(spellId);
        console.log(`${spellId} cooldown expired`);
      } else {
        this.cooldowns.set(spellId, newTurns);
      }
      
      this.notifyListeners(spellId, newTurns);
    });
    
    // Notify about ready spells
    expiredSpells.forEach(spellId => {
      this.onSpellReady(spellId);
    });
  }
  
  /**
   * Called when spell comes off cooldown
   */
  onSpellReady(spellId) {
    // Visual/audio feedback
    this.scene.showMessage(`${SPELL_DATABASE[spellId]?.name || spellId} Ready!`, 0x3CFFD9);
    this.scene.audioManager.play('spell_ready', { volume: 0.5 });
    
    // Pulse spell slot
    this.scene.spellBook.pulseSpellSlot(spellId);
  }
  
  /**
   * Reset all cooldowns (for testing or new combat)
   */
  resetAll() {
    this.cooldowns.clear();
    this.notifyListeners();
  }
  
  /**
   * Subscribe to cooldown changes
   */
  subscribe(callback) {
    this.listeners.push(callback);
  }
  
  notifyListeners(spellId = null, turnsRemaining = 0) {
    this.listeners.forEach(callback => {
      callback(spellId, turnsRemaining);
    });
  }
  
  /**
   * Serialize for save system
   */
  toJSON() {
    return {
      cooldowns: Array.from(this.cooldowns.entries())
    };
  }
  
  /**
   * Deserialize from save
   */
  fromJSON(data) {
    this.cooldowns = new Map(data.cooldowns || []);
  }
}

export default CooldownManager;

        cooldownArc.setVisible(false);
        
        // Store references
        const slotData = {
          slot: slot,
          icon: icon,
          name: name,
          costBadge: costBadge,
          costText: costText,
          cooldownOverlay: cooldownOverlay,
          cooldownText: cooldownText,
          cooldownArc: cooldownArc,
          spell: null,
          index: row * 3 + col
        };
        
        this.spellSlots.push(slotData);
        
        this.container.add(cooldownOverlay);
        this.container.add(cooldownText);
        this.container.add(cooldownArc);
      }
    }
    
    // Subscribe to cooldown updates
    this.scene.cooldownManager.subscribe((spellId, turnsRemaining) => {
      this.updateCooldownDisplay(spellId, turnsRemaining);
    });
  }
  
  updateCooldownDisplay(spellId, turnsRemaining) {
    if (!spellId) {
      // Update all slots
      this.spellSlots.forEach(slot => {
        if (slot.spell) {
          const remaining = this.scene.cooldownManager.getRemainingTurns(slot.spell.id);
          this.updateSlotCooldown(slot, remaining);
        }
      });
      return;
    }
    
    // Find slot with this spell
    const slot = this.spellSlots.find(s => s.spell?.id === spellId);
    if (slot) {
      this.updateSlotCooldown(slot, turnsRemaining);
    }
  }
  
  updateSlotCooldown(slot, turnsRemaining) {
    if (turnsRemaining > 0) {
      // Show cooldown
      slot.cooldownOverlay.setVisible(true);
      slot.cooldownText.setVisible(true);
      slot.cooldownText.setText(turnsRemaining.toString());
      
      // Draw progress arc
      this.drawCooldownArc(slot, turnsRemaining);
      
      // Disable slot interaction
      slot.slot.removeInteractive();
      slot.slot.setAlpha(0.5);
      
    } else {
      // Hide cooldown
      slot.cooldownOverlay.setVisible(false);
      slot.cooldownText.setVisible(false);
      slot.cooldownArc.setVisible(false);
      
      // Re-enable slot
      slot.slot.setInteractive({ useHandCursor: true });
      slot.slot.setAlpha(1);
    }
  }
  
  drawCooldownArc(slot, turnsRemaining) {
    const graphics = slot.cooldownArc;
    graphics.clear();
    graphics.setVisible(true);
    
    if (!slot.spell || !slot.spell.cooldown) return;
    
    const maxCooldown = slot.spell.cooldown;
    const progress = turnsRemaining / maxCooldown;
    
    // Draw arc (clockwise from top)
    const centerX = slot.slot.x;
    const centerY = slot.slot.y;
    const radius = 35;
    const startAngle = -Math.PI / 2;  // Top
    const endAngle = startAngle + (Math.PI * 2 * progress);
    
    graphics.lineStyle(4, 0xFF6B35, 0.8);
    graphics.beginPath();
    graphics.arc(centerX, centerY, radius, startAngle, endAngle, false);
    graphics.strokePath();
    
    // Animate arc shrinking
    this.scene.tweens.add({
      targets: graphics,
      alpha: { from: 1, to: 0.6 },
      duration: 500,
      yoyo: true
    });
  }
  
  pulseSpellSlot(spellId) {
    const slot = this.spellSlots.find(s => s.spell?.id === spellId);
    if (!slot) return;
    
    // Green pulse when spell ready
    const pulse = this.scene.add.circle(slot.slot.x, slot.slot.y, 40, 0x00FF00, 0);
    this.container.add(pulse);
    
    this.scene.tweens.add({
      targets: pulse,
      alpha: 0.6,
      scale: 1.5,
      duration: 300,
      ease: 'Cubic.easeOut',
      onComplete: () => {
        this.scene.tweens.add({
          targets: pulse,
          alpha: 0,
          duration: 300,
          onComplete: () => pulse.destroy()
        });
      }
    });
  }
}

class TurnManager {
  constructor(scene) {
    this.scene = scene;
    this.currentTurn = 0;
    this.turnOrder = [];  // [player, enemy1, enemy2, ...]
    this.activeEntityIndex = 0;
  }
  
  initialize() {
    // Build turn order
    this.turnOrder = [this.scene.player];
    this.turnOrder.push(...this.scene.enemies);
    
    this.currentTurn = 0;
    this.activeEntityIndex = 0;
    
    console.log(`Turn order: ${this.turnOrder.length} entities`);
  }
  
  getCurrentEntity() {
    return this.turnOrder[this.activeEntityIndex];
  }
  
  isPlayerTurn() {
    return this.getCurrentEntity() === this.scene.player;
  }
  
  endTurn() {
    console.log(`Turn ${this.currentTurn} ended for ${this.getCurrentEntity().name}`);
    
    // Move to next entity
    this.activeEntityIndex++;
    
    // If we've gone through all entities, start new round
    if (this.activeEntityIndex >= this.turnOrder.length) {
      this.activeEntityIndex = 0;
      this.currentTurn++;
      
      // Tick cooldowns at start of each full round
      this.scene.cooldownManager.tick();
      
      // Tick status effects
      this.tickStatusEffects();
      
      // Emit round start event
      this.scene.events.emit('round_start', this.currentTurn);
      
      console.log(`=== ROUND ${this.currentTurn} START ===`);
    }
    
    // Start next entity's turn
    this.startTurn();
  }
  
  startTurn() {
    const entity = this.getCurrentEntity();
    console.log(`Turn start: ${entity.name || 'Entity'}`);
    
    // Emit turn start event
    this.scene.events.emit('turn_start', entity);
    
    // If AI entity, run AI logic
    if (entity !== this.scene.player) {
      this.scene.time.delayedCall(500, () => {
        this.runAITurn(entity);
      });
    } else {
      // Player turn - enable input
      this.scene.playerCanAct = true;
    }
  }
  
  runAITurn(entity) {
    // Simple AI: move toward player and attack if in range
    // (This is placeholder - we'll improve AI later)
    
    // End AI turn after 1 second
    this.scene.time.delayedCall(1000, () => {
      this.endTurn();
    });
  }
  
  tickStatusEffects() {
    // Process all timed effects
    this.turnOrder.forEach(entity => {
      // Poison damage
      if (entity.poisoned && entity.poisonDuration > 0) {
        this.scene.dealDamage(entity, entity.poisonDamage);
        entity.poisonDuration--;
        
        if (entity.poisonDuration <= 0) {
          entity.poisoned = false;
          console.log(`${entity.name} poison expired`);
        }
      }
      
      // Immobilize duration
      if (entity.immobilized && entity.immobilizeDuration > 0) {
        entity.immobilizeDuration--;
        
        if (entity.immobilizeDuration <= 0) {
          entity.immobilized = false;
          console.log(`${entity.name} can move again`);
        }
      }
      
      // Buff/debuff durations
      // (Will expand this as we add more status effects)
    });
  }
  
  skipToPlayerTurn() {
    // For testing: immediately jump to player's turn
    this.activeEntityIndex = 0;
    this.startTurn();
  }
}

// Add to GameScene.js

applySpellEffects(spell, targetData) {
  // ... existing Tier 1 spell code ...
  
  // ═══ TIER 2 SPELL EFFECTS ═══
  
  switch (spell.id) {
    case 'soulbind_chains':
      if (targetData.target) {
        // Mark target with soulbind
        targetData.target.soulbound = true;
        targetData.target.soulboundTurns = spell.effects.markDuration;
        targetData.target.soulboundCaster = this.player;
        
        // Visual chains
        const chains = this.particleManager.soulbindChains(
          targetData.target.x,
          targetData.target.y
        );
        targetData.target.soulbindChains = chains;
        
        // Check on death
        targetData.target.onDeath = () => {
          if (targetData.target.soulbound && targetData.target.soulboundTurns > 0) {
            // Resurrect as ally
            this.resurrectAsAlly(targetData.target, spell.effects);
          }
        };
      }
      break;
      
    case 'verdant_surge':
      // Find all allies in radius
      const surgeCenterX = this.player.x;
      const surgeCenterY = this.player.y;
      
      this.allies.forEach(ally => {
        const distance = Phaser.Math.Distance.Between(
          surgeCenterX, surgeCenterY,
          ally.x, ally.y
        ) / 50;
        
        if (distance <= spell.aoeRadius) {
          // Heal
          this.healTarget(ally, spell.effects.healAmount);
          
          // Remove all debuffs
          spell.effects.debuffTypes.forEach(debuffType => {
            if (ally[debuffType]) {
              ally[debuffType] = false;
              this.showMessage(`${debuffType} removed!`, 0x3CFFD9);
            }
          });
        }
      });
      
      // VFX
      this.particleManager.verdantSurge(surgeCenterX, surgeCenterY, spell.aoeRadius);
      break;
      
    case 'sylors_wrath':
      // Start charging (special case: multi-turn spell)
      this.startChanneling(spell, targetData);
      break;
      
    case 'veilkeepers_judgment':
      if (targetData.target) {
        // Show riddle UI
        this.showRiddleChoice(spell, targetData.target);
      }
      break;
      
    case 'mycelium_network':
      // Create network tiles
      const networkTiles = this.calculateNetworkTiles(
        targetData.x,
        targetData.y,
        spell.effects.networkSize
      );
      
      const network = this.particleManager.myceliumNetwork(
        targetData.x,
        targetData.y,
        networkTiles
      );
      
      // Track active network
      this.activeNetworks.push({
        tiles: networkTiles,
        duration: spell.effects.duration,
        graphics: network,
        sharedHP: true
      });
      
      // Calculate shared HP pool
      let totalHP = 0;
      this.allies.forEach(ally => {
        if (this.isOnNetwork(ally, networkTiles)) {
          totalHP += ally.hp;
        }
      });
      
      this.allies.forEach(ally => {
        if (this.isOnNetwork(ally, networkTiles)) {
          ally.networkHP = totalHP;
          ally.onNetwork = true;
        }
      });
      break;
      
    case 'crimson_ascension':
      // Transform player
      this.player.crimsonForm = true;
      this.player.damageBonus = (this.player.damageBonus || 1) * 1.5;
      
      const ascension = this.particleManager.crimsonAscension(
        this.player.x,
        this.player.y
      );
      this.player.crimsonAura = ascension;
      
      // Drain DSP each turn
      this.events.on('turn_start', this.crimsonDrainListener = () => {
        if (this.player.crimsonForm) {
          const result = this.dspManager.consume(spell.effects.dspDrainPerTurn, 'Crimson Form');
          
          if (!result.success) {
            // Form ends when DSP reaches 0
            this.endCrimsonForm();
          }
        }
      });
      
      // Manual cancel button
      this.showCrimsonCancelButton();
      break;
  }
}

// Helper methods for Tier 2 spells

resurrectAsAlly(deadEnemy, effects) {
  console.log(`Resurrecting ${deadEnemy.name} as ally`);
  
  // Soul rebirth VFX
  this.particleManager.soulRebirth(deadEnemy.x, deadEnemy.y);
  
  // Create temporary ally
  const ally = this.add.rectangle(
    deadEnemy.x,
    deadEnemy.y,
    48,
    48,
    0x3CFFD9  // Teal (friendly)
  );
  
  ally.hp = Math.floor(deadEnemy.maxHp * effects.statMultiplier);
  ally.maxHp = ally.hp;
  ally.attack = Math.floor(deadEnemy.attack * effects.statMultiplier);
  ally.name = `Bound ${deadEnemy.name}`;
  ally.isTemporary = true;
  ally.duration = effects.allyDuration;
  
  this.allies.push(ally);
  
  this.showMessage(`${ally.name} fights for you!`, 0x3CFFD9);
  
  // Remove after duration
  this.time.delayedCall(effects.allyDuration * 1000, () => {
    this.tweens.add({
      targets: ally,
      alpha: 0,
      duration: 500,
      onComplete: () => {
        ally.destroy();
        this.allies = this.allies.filter(a => a !== ally);
      }
    });
  });
}

startChanneling(spell, targetData) {
  console.log(`Channeling ${spell.name}...`);
  
  this.player.channeling = {
    spell: spell,
    target: targetData,
    turnsRemaining: spell.effects.chargeTurns,
    currentTurn: 0
  };
  
  // Show charging VFX
  const chargeVFX = this.particleManager.sylorsWrathCharge(
    this.player.x,
    this.player.y,
    1
  );
  this.player.channelVFX = chargeVFX;
  
  // Channel bar UI
  this.showChannelBar(spell);
  
  // Progress each turn
  this.events.on('turn_start', this.channelListener = () => {
    if (this.player.channeling) {
      this.player.channeling.currentTurn++;
      
      // Update VFX
      if (this.player.channelVFX) {
        this.player.channelVFX.destroy();
      }
      this.player.channelVFX = this.particleManager.sylorsWrathCharge(
        this.player.x,
        this.player.y,
        this.player.channeling.currentTurn + 1
      );
      
      // Check if fully charged
      if (this.player.channeling.currentTurn >= spell.effects.chargeTurns) {
        this.releaseChannel();
      }
    }
  });
}

releaseChannel() {
  const channeling = this.player.channeling;
  const spell = channeling.spell;
  const targetData = channeling.target;
  
  console.log(`Releasing ${spell.name}!`);
  
  // Explosion VFX
  this.particleManager.sylorsWrathExplosion(
    targetData.x,
    targetData.y,
    spell.aoeRadius
  );
  
  // Damage all enemies in radius
  this.enemies.forEach(enemy => {
    const distance = Phaser.Math.Distance.Between(
      targetData.x, targetData.y,
      enemy.x, enemy.y
    ) / 50;
    
    if (distance <= spell.aoeRadius) {
      this.dealDamage(enemy, spell.effects.burstDamage);
    }
  });
  
  // Create burning ground
  const burningFlames = this.particleManager.burningGround(
    targetData.x,
    targetData.y,
    spell.aoeRadius
  );
  
  this.burningGrounds.push({
    x: targetData.x,
    y: targetData.y,
    radius: spell.aoeRadius,
    damage: spell.effects.burningDamage,
    duration: spell.effects.burningDuration,
    vfx: burningFlames
  });
  
  // End channeling
  this.player.channeling = null;
  if (this.player.channelVFX) {
    this.player.channelVFX.destroy();
  }
  this.hideChannelBar();
  this.events.off('turn_start', this.channelListener);
}

answerRiddle(spell, target, correct, riddleUI, buttons) {
  // Hide riddle UI
  this.tweens.add({
    targets: [riddleUI.box, riddleUI.text, ...buttons.flatMap(b => [b.button, b.text])],
    alpha: 0,
    duration: 300,
    onComplete: () => {
      riddleUI.box.destroy();
      riddleUI.text.destroy();
      buttons.forEach(b => {
        b.button.destroy();
        b.text.destroy();
      });
    }
  });
  
  if (correct) {
    // Target gains magic immunity
    target.magicImmune = true;
    target.immunityDuration = spell.effects.immunityDuration;

        quantity: 50,
        emitting: false
      });

    });
  }
}

calculateNetworkTiles(centerX, centerY, size) {
  const tiles = [];
  const tileSize = 50;
  
  // Create roughly circular network
  for (let dx = -size; dx <= size; dx++) {
    for (let dy = -size; dy <= size; dy++) {
      const distance = Math.sqrt(dx * dx + dy * dy);
      if (distance <= size) {
        tiles.push({
          x: centerX + (dx * tileSize),
          y: centerY + (dy * tileSize)
        });
      }
    }
  }
  
  return tiles;
}

isOnNetwork(entity, networkTiles) {
  return networkTiles.some(tile => {
    const distance = Phaser.Math.Distance.Between(
      entity.x, entity.y,
      tile.x, tile.y
    );
    return distance < 25;  // Within half a tile
  });
}

  formatSpellStats(spell) {
    let stats = [];
    
    stats.push(`DSP Cost: ${spell.dspCost}`);
    if (spell.cooldown > 0) stats.push(`Cooldown: ${spell.cooldown} turns`);
    if (spell.range > 0) stats.push(`Range: ${spell.range} tiles`);
    if (spell.aoeRadius > 0) stats.push(`AOE: ${spell.aoeRadius} tiles`);
    
    // Spell-specific stats
    if (spell.effects.baseDamage) stats.push(`Damage: ${spell.effects.baseDamage}`);
    if (spell.effects.healAmount) stats.push(`Heal: ${spell.effects.healAmount}`);
    if (spell.effects.defenseBonus) stats.push(`Defense: +${spell.effects.defenseBonus}`);
    if (spell.effects.duration) stats.push(`Duration: ${spell.effects.duration} turns`);
    
    return stats.join('\n');

  formatUpgradedStats(spell, upgrade) {
    let stats = [];
    
    // Same base stats
    stats.push(`DSP Cost: ${spell.dspCost}`);
    if (spell.cooldown > 0) stats.push(`Cooldown: ${spell.cooldown} turns`);
    if (spell.range > 0) stats.push(`Range: ${spell.range} tiles`);
    if (spell.aoeRadius > 0) stats.push(`AOE: ${spell.aoeRadius} tiles`);
    
    // Improved stats (show in green)
    if (upgrade.improvements.baseDamage) {
      stats.push(`Damage: ${upgrade.improvements.baseDamage} ↑`);
    }
    if (upgrade.improvements.healAmount) {
      stats.push(`Heal: ${upgrade.improvements.healAmount} ↑`);
    }
    if (upgrade.improvements.defenseBonus) {
      stats.push(`Defense: +${upgrade.improvements.defenseBonus} ↑`);
    }
    if (upgrade.improvements.duration) {
      stats.push(`Duration: ${upgrade.improvements.duration} turns ↑`);
    }
    
    return stats.join('\n');

  formatNewEffects(upgrade) {
    let effects = ['NEW EFFECTS:'];
    
    Object.entries(upgrade.newEffects).forEach(([key, value]) => {
      // Format effect names nicely
      const effectName = key
        .replace(/([A-Z])/g, ' $1')
        .replace(/^./, str => str.toUpperCase())
        .trim();
      
      effects.push(`• ${effectName}`);
    });
    
    return effects.join('\n');

  formatCost(cost, hasResources) {
    const parts = [];
    
    if (cost.verdantEssence) {
      const has = this.scene.player.verdantEssence || 0;
      const color = has >= cost.verdantEssence ? '✓' : '✗';
      parts.push(`${color} ${cost.verdantEssence} Verdant Essence`);
    }
    
    if (cost.dspInvestment) {
      const has = this.scene.dspManager.maxDSP;
      const color = has >= 100 ? '✓' : '✗';  // Need full DSP pool
      parts.push(`${color} ${cost.dspInvestment} DSP (permanent investment)`);
    }
    
    return 'COST: ' + parts.join(' | ');

    const hasEssence = (this.scene.player.verdantEssence || 0) >= (cost.verdantEssence || 0);
    const hasDSP = this.scene.dspManager.maxDSP >= 100;
    
    return hasEssence && hasDSP;
  }

    const spell = this.selectedSpell;
    const upgrade = SPELL_UPGRADES[spell.id];
    
    // Deduct resources
    this.scene.player.verdantEssence -= upgrade.cost.verdantEssence;
    
    // DSP investment reduces max DSP
    this.scene.dspManager.maxDSP -= upgrade.cost.dspInvestment;
    this.scene.dspManager.currentDSP = Math.min(
      this.scene.dspManager.currentDSP,
      this.scene.dspManager.maxDSP
    );
    
    // Apply upgrade
    this.applyUpgrade(spell, upgrade);
    
    // Visual feedback
    this.scene.showMessage(`${upgrade.upgradeName} UNLOCKED!`, 0x3CFFD9);
    this.scene.audioManager.play('spell_upgrade');

    flash.setScrollFactor(0);

    this.scene.tweens.add({
      targets: flash,
      alpha: 0,
      duration: 500,
      onComplete: () => flash.destroy()
    });
    
    this.close();
  }
  
  applyUpgrade(spell, upgrade) {
    // Mark spell as upgraded
    spell.isUpgraded = true;
    spell.upgradeName = upgrade.upgradeName;
    
    // Apply stat improvements
    Object.entries(upgrade.improvements).forEach(([stat, value]) => {
      if (spell.effects[stat] !== undefined) {
        spell.effects[stat] = value;
      } else {
        spell[stat] = value;
      }
    });
    
    // Add new effects
    Object.entries(upgrade.newEffects).forEach(([effect, value]) => {
      spell.effects[effect] = value;
    });
    
    // Update visual changes
    if (upgrade.visualChanges) {
      spell.visualChanges = upgrade.visualChanges;
    }
    
    console.log(`Upgraded ${spell.name} to ${upgrade.upgradeName}`);
  }
}

```javascript
// src/systems/ComboTracker.js

class ComboTracker {
  constructor(scene) {
    this.scene = scene;
    this.recentCasts = [];  // [{spellId, turn, target}, ...]
    this.activeCombo = null;
    this.comboHistory = [];  // Track triggered combos for stats
  }
  
  /**
   * Record a spell cast
   */
  recordCast(spellId, target, turn) {
    this.recentCasts.push({
      spellId: spellId,
      target: target,
      turn: turn,
      timestamp: Date.now()
    });
    
    // Clean up old casts (keep last 10 turns)
    this.recentCasts = this.recentCasts.filter(cast => {
      return turn - cast.turn <= 10;
    });
    
    // Check for combos
    this.checkForCombos(turn);
  }
  
  /**
   * Check if any combos have been triggered
   */
  checkForCombos(currentTurn) {
    Object.entries(SPELL_COMBOS).forEach(([comboKey, combo]) => {
      if (this.isComboTriggered(combo, currentTurn)) {
        this.triggerCombo(combo);
      }
    });
  }
  
  /**
   * Check if specific combo conditions are met
   */
  isComboTriggered(combo, currentTurn) {
    const requiredSpells = combo.spells;
    const windowTurns = combo.windowTurns;
    
    // Find casts of required spells within window
    const relevantCasts = this.recentCasts.filter(cast => {
      const withinWindow = currentTurn - cast.turn <= windowTurns;
      const isRequiredSpell = requiredSpells.includes(cast.spellId);
      return withinWindow && isRequiredSpell;
    });
    
    // Check if we have all required spells
    if (requiredSpells.length === 1) {
      // Special case: same spell multiple times
      const spellId = requiredSpells[0];
      const castCount = relevantCasts.filter(c => c.spellId === spellId).length;
      return castCount >= requiredSpells.length;
    }
    
    const uniqueCasts = new Set(relevantCasts.map(c => c.spellId));
    if (uniqueCasts.size < requiredSpells.length) {
      return false;  // Missing spells
    }
    
    // Check order if strict
    if (combo.order === 'strict') {
      return this.checkStrictOrder(requiredSpells, relevantCasts);
    }
    
    return true;  // All spells cast, order doesn't matter
  }
  
  checkStrictOrder(requiredSpells, casts) {
    // Sort casts by turn
    const sortedCasts = casts.sort((a, b) => a.turn - b.turn);
    
    let requiredIndex = 0;
    for (let cast of sortedCasts) {
      if (cast.spellId === requiredSpells[requiredIndex]) {
        requiredIndex++;
      }
      
      if (requiredIndex >= requiredSpells.length) {
        return true;  // Found all in correct order
      }
    }
    
    return false;  // Didn't find correct order
  }
  
  /**
   * Execute combo effects
   */
  triggerCombo(combo) {
    console.log(`COMBO TRIGGERED: ${combo.name}`);
    
    // Prevent triggering same combo twice in a row
    if (this.activeCombo === combo.name) {
      return;
    }
    
    this.activeCombo = combo.name;
    this.comboHistory.push({
      name: combo.name,
      turn: this.scene.turnManager.currentTurn
    });
    
    // Visual/audio feedback
    this.scene.showComboNotification(combo);
    this.scene.audioManager.play(combo.soundEffect || 'combo_generic');
    
    // Apply combo effect
    this.applyComboEffect(combo);
    
    // Reset active combo after 1 second
    this.scene.time.delayedCall(1000, () => {
      this.activeCombo = null;
    });
  }
  
  applyComboEffect(combo) {
    const effect = combo.effect;
    
    switch (effect.type) {
      case 'damage_amplify':
        this.applyDamageAmplify(effect);
        break;
        
      case 'damage_bonus':
        this.applyDamageBonus(effect);
        break;
        
      case 'defense_synergy':
        this.applyDefenseSynergy(effect);
        break;
        
      case 'heal_amplify':
        this.applyHealAmplify(effect);
        break;
        
      case 'immobilize_plus':
        this.applyImmobilizePlus(effect);
        break;
        
      case 'nuclear_option':
        this.applyNuclearOption(effect);
        break;
        
      case 'network_amplify':
        this.applyNetworkAmplify(effect);
        break;
        
      case 'auto_correct':
        this.applyAutoCorrect(effect);
        break;
        
      case 'rapid_fire':
        this.applyRapidFire(effect);
        break;
        
      case 'healing_chain':
        this.applyHealingChain(effect);
        break;
    }
  }
  
  applyDamageAmplify(effect) {
    // Find recently applied poison effects
    this.scene.enemies.forEach(enemy => {
      if (enemy.poisoned && enemy.immobilized) {
        // This is the combo target
        enemy.poisonDamage *= effect.multiplier;
        
        // Enhanced VFX
        this.scene.particleManager.toxicRootCombo(enemy.x, enemy.y);
        
        this.scene.showMessage(effect.message, 0x88CC66);
      }
    });
  }
  
  applyDamageBonus(effect) {
    // Next attack gets bonus
    this.scene.player.nextAttackBonus = (this.scene.player.nextAttackBonus || 0) + effect.bonusDamage;
    
    // Momentum VFX
    this.scene.particleManager.momentumBlur(this.scene.player.x, this.scene.player.y);
    
    this.scene.showMessage(effect.message, 0xFFAA00);
  }
  
  applyDefenseSynergy(effect) {
    // Enhance active shield
    if (this.scene.player.shield) {
      this.scene.player.defenseBonus += effect.shieldBonus;
    }
    
    // Enhance active barrier
    this.scene.rootBarriers.forEach(barrier => {
      barrier.reflectDamage = effect.barrierReflect;
    });
    
    this.scene.showMessage(effect.message, 0x3CFFD9);
  }
  
  applyHealAmplify(effect) {
    // Enhance active blooms
    this.scene.restorativeBlooms.forEach(bloom => {
      bloom.healPerTurn = Math.floor(bloom.healPerTurn * effect.healMultiplier);
      bloom.grantsCrit = effect.grantsCritBonus;
      
      // Enhanced VFX
      bloom.sprite.setScale(1.5);
      bloom.sprite.setTint(0xFFD700);
    });
    
    this.scene.showMessage(effect.message, 0x00FF00);
  }
  
  applyImmobilizePlus(effect) {
    // Enhance active traps
    this.scene.activeTraps.forEach(trap => {
      trap.duration = effect.trapDuration;
      trap.guaranteedSoulbind = effect.soulbindGuaranteed;
    });
    
    this.scene.showMessage(effect.message, 0xCCFFFF);
  }
  
  applyNuclearOption(effect) {
    // Store combo state for when Wrath is cast
    this.scene.crimsonWrathCombo = {
      active: true,
      instantCast: effect.wrathInstantCast,
      damage: effect.wrathDamage,
      radius: effect.wrathRadius,
      costAllDSP: effect.costAllDSP
    };
    
    this.scene.showMessage(effect.message, 0xFF0000);
    
    // Screen turns red
    const redTint = this.scene.add.rectangle(400, 300, 800, 600, 0xFF0000, 0.3);
    redTint.setScrollFactor(0);
    redTint.setDepth(1500);
    
    this.scene.tweens.add({
      targets: redTint,
      alpha: 0,
      duration: 2000,
      onComplete: () => redTint.destroy()
    });
  }
  
  applyNetworkAmplify(effect) {
    this.scene.networkSurgeCombo = {
      active: true,
      freeSurge: effect.surgeCostZero,
      noCooldown: effect.surgeInstantCooldown,
      radius: effect.surgeRadius
    };
    
    this.scene.showMessage(effect.message, 0x88CC66);
  }
  
  applyAutoCorrect(effect) {
    this.scene.shadowJudgmentCombo = {
      active: true,
      autoFail: effect.forcedCorrectAnswer,
      maintainStealth: effect.stealthMaintained
    };
    
    this.scene.showMessage(effect.message, 0x6666FF);
  }
  
  applyRapidFire(effect) {
    // Track bolts cast this turn
    if (!this.scene.viridianBoltsThisTurn) {
      this.scene.viridianBoltsThisTurn = 0;
    }
    
    this.scene.viridianBoltsThisTurn++;
    
    if (this.scene.viridianBoltsThisTurn >= 3) {
      // Grant rapid fire bonuses
      this.scene.crystalStormActive = true;
      this.scene.showMessage(effect.message, 0x3CFFD9);
    }
  }
  
  applyHealingChain(effect) {
    this.scene.lifebloodLoopCombo = {
      active: true,
      freeSecondCast: effect.secondCastFree,
      chainHealing: effect.chainHealing
    };
    
    this.scene.showMessage(effect.message, 0x00FFCC);
  }
  
  /**
   * Reset turn-based tracking
   */
  resetTurnTracking() {
    this.scene.viridianBoltsThisTurn = 0;
  }
  
  /**
   * Get combo stats for UI
   */
  getComboStats() {

      total: this.comboHistory.length,
      unique: new Set(this.comboHistory.map(c => c.name)).size,
      recent: this.comboHistory.slice(-5)
```

  /**
   * Start a new phase
   */
  startPhase(phase) {
    this.currentPhase = phase;
    this.phaseTurn = 0;
    
    // Randomize duration within range
    const range = this.durations[phase];
    this.phaseDuration = Phaser.Math.Between(range.min, range.max);
    
    console.log(`=== SAP CYCLE: ${phase.toUpperCase()} PHASE ===`);
    console.log(`Duration: ${this.phaseDuration} turns`);
    
    // Apply phase modifiers
    this.applyPhaseModifiers(phase);
    
    // Notify listeners
    this.notifyPhaseChange(phase);
    
    // Visual/audio feedback
    this.scene.showPhaseTransition(phase);
    this.scene.audioManager.play(`sap_phase_${phase}`);
    
    // Special phase logic
    switch (phase) {
      case this.phases.BLUE:
        this.onBluePhaseStart();
        break;
      case this.phases.CRIMSON:
        this.onCrimsonPhaseStart();
        break;
      case this.phases.SILVER:
        this.onSilverPhaseStart();
        break;
    }
  }
  
  /**
   * Tick the cycle (called each turn)
   */
  tick() {
    this.phaseTurn++;
    
    console.log(`Sap Cycle: ${this.currentPhase} (${this.phaseTurn}/${this.phaseDuration})`);
    
    // Check for phase transition
    if (this.phaseTurn >= this.phaseDuration) {
      this.transitionToNextPhase();
    } else {
      // Check for warnings
      this.checkWarnings();
      
      // Apply per-turn effects
      this.applyPerTurnEffects();
    }
  }
  
  /**
   * Transition to next phase in cycle
   */
  transitionToNextPhase() {
    let nextPhase;
    
    switch (this.currentPhase) {
      case this.phases.BLUE:
        nextPhase = this.phases.CRIMSON;
        break;
      case this.phases.CRIMSON:
        nextPhase = this.phases.SILVER;
        break;
      case this.phases.SILVER:
        nextPhase = this.phases.BLUE;
        this.totalCycles++;
        console.log(`Completed cycle #${this.totalCycles}`);
        break;
    }
    
    this.startPhase(nextPhase);
  }
  
  /**
   * Apply modifiers for current phase
   */
  applyPhaseModifiers(phase) {
    const mods = this.modifiers[phase];
    
    // Store modifiers on scene for easy access
    this.scene.sapPhaseModifiers = mods;
    
    // Update UI
    this.scene.updatePhaseUI(phase, mods);
  }
  
  /**
   * Apply per-turn effects
   */
  applyPerTurnEffects() {
    const mods = this.modifiers[this.currentPhase];
    
    switch (this.currentPhase) {
      case this.phases.BLUE:
        // HP regeneration for player
        if (this.scene.player.hp < this.scene.player.maxHp) {
          const regen = Math.floor(this.scene.player.maxHp * mods.hpRegenBonus);
          if (regen > 0) {
            this.scene.healTarget(this.scene.player, regen);
            this.scene.showMessage(`+${regen} HP (Blue Sap)`, 0x3CFFD9);
          }
        }
        break;
        
      case this.phases.SILVER:
        // DSP regeneration
        if (mods.dspRegeneration) {
          this.scene.dspManager.restore(mods.dspRegeneration, 'Silver Sap');
        }
        break;
    }
  }
  
  /**
   * Check for phase transition warnings
   */
  checkWarnings() {
    const remaining = this.phaseDuration - this.phaseTurn;
    
    // Warning at 5 turns remaining
    if (remaining === 5) {
      this.notifyWarning('phase_ending', remaining);
      this.scene.showMessage(`${this.currentPhase.toUpperCase()} phase ending in ${remaining} turns!`, 0xFF6B35);
    }
    
    // Warning at 1 turn remaining
    if (remaining === 1) {
      this.notifyWarning('phase_imminent', remaining);
      this.scene.showMessage(`Phase transition imminent!`, 0xFF0000);
    }
    
    // Crimson-specific warnings (Veilkeeper whispers)
    if (this.currentPhase === this.phases.CRIMSON && remaining <= 3) {
      this.veilkeeperWhisper(`The silver tide approaches. Plant your seeds while you can.`);
    }
  }
  
  /**
   * Blue Phase logic
   */
  onBluePhaseStart() {
    console.log('Blue Sap: Growth phase begins');
    
    // Spawn new environmental features
    this.scene.spawnBluePhaseCreatures();
    
    // Visual changes
    this.scene.setEnvironmentTint(0xCCEEFF);  // Blue tint
    this.scene.addBlueGlowToTrees();
  }
  
  /**
   * Crimson Phase logic
   */
  onCrimsonPhaseStart() {
    console.log('Crimson Sap: Planting phase begins');
    
    // Enable planting UI
    this.scene.showPlantingUI();
    
    // Enemies become more aggressive
    this.scene.enemies.forEach(enemy => {
      enemy.aggressionMultiplier = 1.3;
    });
    
    // Visual changes
    this.scene.setEnvironmentTint(0xFFCCCC);  // Crimson tint
    this.scene.addCrimsonPulseToTrees();
    
    // Veilkeeper warning
    this.veilkeeperWhisper('The sap runs crimson. Plant wisely, for the Unbinding comes.');
  }
  
  /**
   * Silver Phase logic
   */
  onSilverPhaseStart() {
    console.log('Silver Sap: THE UNBINDING!');
    
    // Screen shake
    this.scene.cameras.main.shake(500, 0.01);
    
    // Massive VFX
    this.scene.triggerUnbindingVFX();
    
    // Unbind all planted seeds
    this.unbindSeeds();
    
    // Visual changes
    this.scene.setEnvironmentTint(0xFFFFFF);  // Bright white/silver
    this.scene.addSilverExplosionToTrees();
    
    // All spells empowered
    this.scene.showMessage('ALL SPELLS EMPOWERED!', 0xFFFFFF);
  }
  
  /**
   * Plant a seed during Crimson phase
   */
  plantSeed(x, y, type = 'basic') {
    if (this.currentPhase !== this.phases.CRIMSON) {
      console.warn('Can only plant during Crimson phase');
      return false;
    }
    
    const seed = {
      x: x,
      y: y,
      type: type,
      plantedTurn: this.scene.turnManager.currentTurn,
      sprite: this.scene.createSeedSprite(x, y, type)
    };
    
    this.plantedSeeds.push(seed);
    
    console.log(`Planted ${type} seed at (${x}, ${y})`);
    this.scene.showMessage('Seed planted!', 0xFF6B35);
    
    return true;
  }
  
  /**
   * Unbind seeds (Silver phase)
   */
  unbindSeeds() {
    console.log(`Unbinding ${this.plantedSeeds.length} seeds...`);
    
    this.plantedSeeds.forEach((seed, index) => {
      // Delay each unbinding slightly for dramatic effect
      this.scene.time.delayedCall(index * 200, () => {
        this.unbindSingleSeed(seed);
      });
    });
    
    // Clear planted seeds
    this.plantedSeeds = [];
  }
  
  unbindSingleSeed(seed) {
    // Dramatic VFX
    this.scene.particleManager.seedUnbinding(seed.x, seed.y);
    
    // Destroy seed sprite
    if (seed.sprite) {
      this.scene.tweens.add({
        targets: seed.sprite,
        alpha: 0,
        scale: 2,
        duration: 500,
        onComplete: () => seed.sprite.destroy()
      });
    }
    
    // Spawn creature/ally based on seed type
    const spawned = this.scene.spawnFromSeed(seed);
    
    console.log(`Seed unbound → spawned ${spawned.type}`);
  }
  
  /**
   * Veilkeeper whisper (ambient story)
   */
  veilkeeperWhisper(message) {
    this.scene.showVeilkeeperWhisper(message);
    this.scene.audioManager.play('veilkeeper_whisper', { volume: 0.7 });
  }
  
  /**
   * Get current phase info
   */
  getCurrentPhaseInfo() {
    return {
      phase: this.currentPhase,
      turn: this.phaseTurn,
      duration: this.phaseDuration,
      remaining: this.phaseDuration - this.phaseTurn,
      modifiers: this.modifiers[this.currentPhase],
      totalCycles: this.totalCycles
    };
  }
  
  /**
   * Subscribe to phase changes
   */
  onPhaseChange(callback) {
    this.phaseChangeListeners.push(callback);
  }
  
  notifyPhaseChange(phase) {
    this.phaseChangeListeners.forEach(callback => {
      callback(phase, this.modifiers[phase]);
    });
  }
  
  /**
   * Subscribe to warnings
   */
  onWarning(callback) {
    this.warningListeners.push(callback);
  }
  
  notifyWarning(type, data) {
    this.warningListeners.forEach(callback => {
      callback(type, data);
    });
  }
  
  /**
   * Serialize for save system
   */
  toJSON() {
    return {
      currentPhase: this.currentPhase,
      phaseTurn: this.phaseTurn,
      phaseDuration: this.phaseDuration,
      totalCycles: this.totalCycles,
      plantedSeeds: this.plantedSeeds.map(seed => ({
        x: seed.x,
        y: seed.y,
        type: seed.type,
        plantedTurn: seed.plantedTurn
      }))
    };
  }
  
  /**
   * Deserialize from save
   */
  fromJSON(data) {
    this.currentPhase = data.currentPhase;
    this.phaseTurn = data.phaseTurn;
    this.phaseDuration = data.phaseDuration;
    this.totalCycles = data.totalCycles;
    
    // Restore planted seeds
    this.plantedSeeds = data.plantedSeeds.map(seedData => {
      return {
        x: seedData.x,
        y: seedData.y,
        type: seedData.type,
        plantedTurn: seedData.plantedTurn,
        sprite: this.scene.createSeedSprite(seedData.x, seedData.y, seedData.type)
      };
    });
    
    // Reapply current phase modifiers
    this.applyPhaseModifiers(this.currentPhase);
  }
}

// Add to GameScene.js

setEnvironmentTint(color) {
  // Store original colors if not already stored
  if (!this.originalTints) {
    this.originalTints = new Map();
  }
  
  // Tint all terrain tiles
  this.terrainTiles.forEach(tile => {
    if (!this.originalTints.has(tile)) {
      this.originalTints.set(tile, tile.fillColor);
    }
    
    // Blend original color with tint
    const blendedColor = this.blendColors(this.originalTints.get(tile), color, 0.3);
    tile.setFillStyle(blendedColor);
  });
  
  // Tint trees
  this.trees.forEach(tree => {
    if (!this.originalTints.has(tree)) {
      this.originalTints.set(tree, tree.fillColor);
    }
    
    const blendedColor = this.blendColors(this.originalTints.get(tree), color, 0.4);
    tree.setFillStyle(blendedColor);
  });
  
  // Ambient particles
  this.updateAmbientParticles(color);
}

blendColors(color1, color2, alpha) {
  // Extract RGB components
  const r1 = (color1 >> 16) & 0xFF;
  const g1 = (color1 >> 8) & 0xFF;
  const b1 = color1 & 0xFF;
  
  const r2 = (color2 >> 16) & 0xFF;
  const g2 = (color2 >> 8) & 0xFF;
  const b2 = color2 & 0xFF;
  
  // Blend
  const r = Math.floor(r1 * (1 - alpha) + r2 * alpha);
  const g = Math.floor(g1 * (1 - alpha) + g2 * alpha);
  const b = Math.floor(b1 * (1 - alpha) + b2 * alpha);
  
  // Combine back to hex
  return (r << 16) + (g << 8) + b;
}

updateAmbientParticles(color) {
  // Kill existing ambient particles
  if (this.ambientParticles) {
    this.ambientParticles.destroy();
  }
  
  // Create new particles with phase color
  this.ambientParticles = this.add.particles(400, 300, 'emerald_particle', {
    speed: { min: 10, max: 30 },
    angle: { min: 0, max: 360 },
    scale: { start: 0.3, end: 0 },
    tint: color,
    alpha: { start: 0.4, end: 0 },
    lifespan: 3000,
    frequency: 200,
    quantity: 1,
    emitZone: {
      type: 'random',
      source: new Phaser.Geom.Rectangle(0, 0, 800, 600)
    }
  });
  this.ambientParticles.setScrollFactor(0.5);  // Parallax effect
  this.ambientParticles.setDepth(50);
}

  selectSeed(slotData) {
    const seedType = slotData.seedType;
    const dsp = this.scene.dspManager.currentDSP;
    
    // Check if can afford
    if (seedType.cost > dsp) {
      this.scene.showMessage('Not enough DSP!', 0xFF6B35);
      return;
    }
    
    // Deselect previous
    if (this.selectedSeedType) {
      const prevSlot = this.seedSlots.find(s => s.seedType === this.selectedSeedType);
      if (prevSlot) {
        prevSlot.slot.setStrokeStyle(2, 0x1a1a1a);
      }
    }
    
    // Select new
    this.selectedSeedType = seedType;
    slotData.slot.setStrokeStyle(3, 0xFF6B35);
    
    // Update details
    this.detailName.setText(seedType.name);
    this.detailDesc.setText(seedType.description);
    this.detailCost.setText(`Cost: ${seedType.cost} DSP`);
    
    // Update plant button
    if (seedType.cost > dsp) {
      this.plantButton.setFillStyle(0x555555);
      this.plantButtonText.setColor('#888888');
      this.plantButton.disableInteractive();
    } else {
      this.plantButton.setFillStyle(0xFF6B35);
      this.plantButtonText.setColor('#FFFFFF');
      this.plantButton.setInteractive({ useHandCursor: true });
    }
  }
  
  updateSeedAvailability() {
    const dsp = this.scene.dspManager.currentDSP;
    
    this.seedSlots.forEach(slotData => {
      if (slotData.seedType.cost > dsp) {
        slotData.slot.setAlpha(0.5);
        slotData.icon.setAlpha(0.5);
        slotData.name.setAlpha(0.5);
      } else {
        slotData.slot.setAlpha(1);
        slotData.icon.setAlpha(1);
        slotData.name.setAlpha(1);
      }
    });
  }
  
  confirmPlant() {
    if (!this.selectedSeedType || !this.plantLocation) return;
    
    // Consume DSP
    const result = this.scene.dspManager.consume(
      this.selectedSeedType.cost,
      `Plant ${this.selectedSeedType.name}`
    );
    
    if (!result.success) {
      this.scene.showMessage('Not enough DSP!', 0xFF6B35);
      return;
    }
    
    // Plant seed
    this.scene.sapCycleManager.plantSeed(
      this.plantLocation.x,
      this.plantLocation.y,
      this.selectedSeedType.id
    );
    
    this.close();
  }

// Add to SapCycleManager.js

plantSeed(x, y, type = 'basic') {
  if (this.currentPhase !== this.phases.CRIMSON) {
    console.warn('Can only plant during Crimson phase');
    return false;
  }
  
  const seedType = SEED_TYPES[type.toUpperCase()];
  if (!seedType) {
    console.error(`Unknown seed type: ${type}`);
    return false;
  }
  
  // Check placement validity
  const validation = this.validatePlacement(x, y, seedType);
  if (!validation.valid) {
    this.scene.showMessage(validation.reason, 0xFF0000);
    return false;
  }
  
  // Create seed
  const seed = {
    x: x,
    y: y,
    type: type,
    seedType: seedType,
    plantedTurn: this.scene.turnManager.currentTurn,
    sprite: this.scene.createSeedSprite(x, y, seedType)
  };
  
  this.plantedSeeds.push(seed);
  
  console.log(`Planted ${seedType.name} at (${x}, ${y})`);
  this.scene.showMessage(`${seedType.name} planted!`, 0xFF6B35);
  
  // Show placement preview
  this.showPlacementPreview(seed);
  
  return true;
}

validatePlacement(x, y, seedType) {
  // Check if spot already has seed
  const existing = this.plantedSeeds.find(seed => {
    const distance = Phaser.Math.Distance.Between(seed.x, seed.y, x, y);
    return distance < 25;
  });
  
  if (existing) {
    return { valid: false, reason: 'Seed already planted here!' };
  }
  
  // Check if on walkable terrain
  if (!this.scene.isWalkable(x, y)) {
    return { valid: false, reason: 'Cannot plant on obstacle!' };
  }
  
  // Check if too close to enemies (for safety)
  const nearbyEnemy = this.scene.enemies.find(enemy => {
    const distance = Phaser.Math.Distance.Between(enemy.x, enemy.y, x, y);
    return distance < 100;  // 2 tiles
  });
  
  if (nearbyEnemy) {
    return { valid: false, reason: 'Too close to enemies!' };
  }
  
  return { valid: true };
}

showPlacementPreview(seed) {
  // Show ghost of what will spawn
  const preview = this.scene.add.rectangle(
    seed.x,
    seed.y,
    48,
    48,
    seed.seedType.color,
    0.3
  );
  preview.setStrokeStyle(2, seed.seedType.color, 0.6);
  
  // Floating text showing what spawns
  const label = this.scene.add.text(
    seed.x,
    seed.y - 30,
    `→ ${seed.seedType.spawns.template}`,
    {
      fontSize: '10px',
      color: Phaser.Display.Color.IntegerToColor(seed.seedType.color).rgba,
      backgroundColor: '#000000',
      padding: { x: 3, y: 2 }
    }
  );
  label.setOrigin(0.5);
  label.setAlpha(0.7);
  
  seed.preview = { sprite: preview, label: label };
}
